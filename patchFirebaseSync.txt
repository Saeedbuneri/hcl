const { getDoc, updateDoc } = require('firebase/firestore');

async function fetchManualSyncDetails(patientId) {
    try {
        let firebaseData = null;
        let localPatient = null;
        let localBookings = [];

        // 1. Fetch Local
        const patientRow = await dbLocal.getPatientById(patientId);
        if (patientRow) {
            localPatient = patientRow;
            localBookings = await dbLocal.getBookingsByPatientId(patientId);
        }

        // 2. Fetch Firebase
        const docRef = doc(fsDb, "reports", patientId);
        const docSnap = await getDoc(docRef);
        if (docSnap.exists()) {
            firebaseData = docSnap.data();
        }

        return { success: true, localPatient, localBookings, firebaseData };
    } catch(e) {
        return { success: false, error: e.message };
    }
}

async function updateManualSyncDetails(patientId, updates) {
    try {
        // updates contains: name, age, gender, phone, pin(password)
        // 1. Update local
        await dbLocal.updatePatientDetailsAndPin(patientId, updates);
        
        // 2. Update Firebase
        const docRef = doc(fsDb, "reports", patientId);
        const docSnap = await getDoc(docRef);
        if (docSnap.exists()) {
            const finalPayload = {
                patient_name: updates.name,
                age: updates.age,
                gender: updates.gender,
                contact: updates.phone,
                password: updates.pin
            };
            await updateDoc(docRef, finalPayload);
        } else {
            // Patient might not be in firebase yet, create basic structure
            const finalPayload = {
                patient_id: patientId,
                patient_name: updates.name,
                age: updates.age,
                gender: updates.gender,
                contact: updates.phone,
                password: updates.pin
            };
            await setDoc(doc(fsDb, "reports", patientId), finalPayload, { merge: true });
        }
        
        // Update user auth optionally (Firebase Auth changing password is a bit harder from client admin without re-auth. I will skip the Auth account update if it's too much, but the \password\ field is the PIN used in online site anyway.)
        return { success: true };
    } catch (e) {
        return { success: false, error: e.message };
    }
}
